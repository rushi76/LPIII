pragma solidity 0.8.0; 
 
contract StudentDatabase { 
    struct Student { 
        string name; 
        uint256 rollNumber; 
        string class; 
    } 
 
    Student[] private students; 
 
    function addStudent(string memory name, uint256 rollNumber, string memory class) 
public { 
        students.push(Student(name, rollNumber, class)); 
    } 
 
    function getStudentByID(uint256 ID) public view returns (string memory, uint256, string 
memory) { 
        require(ID < students.length, "Student does not exist in database"); 
        return (students[ID].name, students[ID].rollNumber, students[ID].class); 
    } 
 
    function getTotalNumberOfStudents() public view returns (uint256) { 
        return students.length; 
    } 
}
